checksudo() {
# Check if "sudo" is installed
    if ! command -v sudo &> /dev/null; then
    printf "\n❌ Error: 'sudo' is not installed.\n\n"
    printf "\n📍 Please log in as 'root' and install it with: > apt update && apt upgrade && apt install sudo\n\n"
    return 1
  fi
}

bootconfig() {
    printf "\n🚀 Optimizing boot configuration for USB mode...\n"
    
    local bootconf="/boot/firmware/config.txt"
    # Fallback for older OS versions
    [ ! -f "$bootconf" ] && bootconf="/boot/config.txt"

    # CHECK: Is the entry already present?
    if grep -q "program_usb_boot_mode=1" "$bootconf"; then
        printf "\nℹ️ Boot parameters are already configured in %s.\n\n" "$bootconf"
        return 0
    else
        printf "\n➕ Adding boot parameters...\n\n"
    if sudo bash -c "cat >> $bootconf" <<-EOF > /dev/null 2>&1
--- USB Boot & Power Tuning ---
# WLAN deativieren, da Server nicht über WLAN laufen sollten !!!
dtoverlay=disable-wifi
# Bluetooth deaktivieren
dtoverlay=disable-bt
# Booten auch von USB
program_usb_boot_mode=1
# mehr Power für USB-Ports
max_usb_current=1
# Startverzögerung von 15 Sekunden.
boot_delay=15 
program_usb_boot_timeout=1
EOF
    then
    printf "\n✅ Boot parameters added successfully.\n\n"
    return 1
     else
     printf "\n❌ Error: Boot parameters could not be saved!\n\n"
     return 2 # ERROR: Writing failed!
    fi
  fi
}

expandfilesystem() {
    printf "\n🚀 Checking disk for unused storage space...\n"
    # 1. Check: Is the required tool available?
    if ! command -v growpart >/dev/null 2>&1; then
        printf "\n❌ Error: The filesystem could not be expanded.\n\n"
        printf "\n👉 The package 'cloud-guest-utils' is not installed.\n"
        printf "\n👉 Please install it manually with: sudo apt install cloud-guest-utils\n"
        return 1
    fi

    # 2. Determine root device and partition (Safe method for Trixie)
    local root_dev=$(findmnt -n -o SOURCE /)
    # Extract parent_dev without /dev/ path
    local parent_dev=$(lsblk -no PKNAME "$root_dev" | head -n 1)
    # Get partition number directly from the kernel system (always works)
    local dev_node=$(basename "$root_dev")
    local part_num=$(cat /sys/class/block/"$dev_node"/partition 2>/dev/null)

    if [ -z "$parent_dev" ] || [ -z "$part_num" ]; then
        printf "⚠️ The drive structure could not be determined. Expansion aborted.\n"
        return 1
    fi

    # 3. Try to enlarge the partition
    printf "\n💾 Expanding filesystem to maximum size...\n\n"
    
    # growpart needs disk (/dev/sda) and number (2) separately
    if sudo growpart "/dev/$parent_dev" "$part_num" > /dev/null 2>&1; then
        # 4. Enlarge filesystem online
        if sudo resize2fs "$root_dev" > /dev/null 2>&1; then
            printf "✅ Success: The partition has been expanded to full size.\n"
        else
            printf "❌ Error: Partition expanded, but filesystem resize failed.\n"
        fi
    else
        printf "ℹ️ Note: The partition already uses all available space.\n"
    fi
}

checkosversion() {
    # 1. Extract the pure version number (e.g. 11 or 12 etc.)
    # tr -d '"' removes possible quotation marks around the number
    local os_ver
    os_ver=$(grep "^VERSION_ID=" /etc/os-release | cut -d= -f2 | tr -d '"')

    printf "\n🔍 Checking operating system version...\n"

    # If the variable is empty (safety check)
    if [[ -z "$os_ver" ]]; then
        printf "\n❌ Error: The OS version could not be determined. - Aborting!\n\n"
        return 1
    fi

    # 2. Comparison: must be version 13 (Trixi) or higher
    if [ "$os_ver" -ge 13 ] 2>/dev/null; then
        printf "\n✅ System version %s detected (OK).\n\n" "$os_ver" 
        # ========================================================
        local_lang_en # Set language to English
        local lang_changed=$?
        bootconfig # Add additional boot parameters
        local boot_changed=$?
          # Restart if necessary
          if [ $lang_changed -ne 0 ] || [ $boot_changed -ne 0 ]; then
          printf "\n🚀 The system must be restarted to apply the changes.\n\n"
          printf "\n🔄 Restarting in 10 seconds (cancel with Ctrl+C)...\n\n"
          sleep 10
          sudo reboot
          return 1
          fi
       return 0
    else
        printf "\n❌ Error: This function set requires at least version 13 (Trixi) of the operating system.\n\n"
        printf "\n⚠️ The current version is: %s\n\n" "$os_ver"
        printf "\n🚫 Unfortunately, we must abort at this point.\n\n"
        return 1
    fi
}

checkpartitionsize() {
    printf "\n📊 Checking available storage space...\n\n"
    
    # Function to determine size
    get_size_mb() {
        # awk automatically removes leading spaces
        df -m / --output=size | tail -1 | awk '{print $1}'
    }

    local root_size_mb=$(get_size_mb)
    local min_size_mb=15360 # 15 GB in MB

    # 2. Try expansion if too small
    if [ "$root_size_mb" -lt "$min_size_mb" ]; then
        # Convert MB to GB for display
        local root_size_display=$((root_size_mb / 1024))
        
        printf "\nℹ️ The partition is too small for Pi-hole (%s GB). Attempting expansion...\n\n" "$root_size_display"
        expandfilesystem 
        
        # Reload size after expansion
        root_size_mb=$(get_size_mb)
    fi

    # 3. Final check
    local root_size_gb=$((root_size_mb / 1024))

    if [ "$root_size_mb" -lt "$min_size_mb" ]; then
        printf "\n⚠️ WARNING: The partition is too small (%s GB).\n\n" "$root_size_gb"
        printf "\n❌ Pi-hole requires at least 16 GB of storage for stable operation.\n\n"
        printf "\n🛑 The installation of Pi-hole will be aborted!\n\n"
        return 1
    else
        printf "\n✅ Available storage space is OK: %s GB detected.\n\n" "$root_size_gb"
        printf "\n⌨️ Press any key to continue...\n\n"
        read -n 1 -s -r
        clear # Clear screen
        return 0
    fi
}

setrouterconfig() {
   local param="$1"
   local router_name
   router_name=$(getroutermodel)

     # Get the IPv4 address
    if [[ "$param" == "-phi" ]]; then
     # Read global IPv6 address if present
     if ip -6 addr show 2>/dev/null | grep -q "scope global"; then
       printf "\n============================================================================================================\n"
       printf "ℹ️ Note: The IPv4 address and the global IPv6 address must now be entered as the DNS server in the router.\n" 
       printf "============================================================================================================\n\n"
     else
        printf "\n=================================================================================\n"
       printf "ℹ️ Note: The IPv4 address must now be entered as the DNS server in the router.\n" 
       printf "=================================================================================\n\n"  
     fi
     printf "\n⌨️ Press any key to continue...\n\n"
     read -n 1 -s -r
    fi
     clear # Clear screen
     local IPv4
     IPv4=$(hostname -I | awk '{print $1}')
      if [[ -n "$router_name" ]]; then
      printf "\n\n🌐 *** Router settings (%s) ***\n\n\n" "$router_name"
      else
      printf "\n\n🌐 *** Router settings (your network) ***\n\n\n"
      fi
        # Formatted output with fixed width 
        printf "###############################################\n"
        printf "########### IPv4 - Router Settings ############\n"
        printf "###############################################\n"
        printf "📝 %-25s: %s\n" "Preferred DNSv4 server" "$IPv4"
        printf "📝 %-25s: %s\n" "Alternative DNSv4 server" "1.1.1.1"
        printf -- "-----------------------------------------------\n\n\n"
     if ip -6 addr show 2>/dev/null | grep -q "scope global"; then
     # Read global IPv6 address if present
     local IPv6
     IPv6=$(ip -6 addr show | grep "scope global" | awk '{print $2}' | cut -d/ -f1 | head -n 1)
     # Formatted output with fixed width 
     printf "########################################################################\n"
     printf "#################### IPv6 - Router Settings ############################\n"
     printf "########################################################################\n"
     printf "📝 %-25s: %s\n" "Preferred DNSv6 server" "$IPv6"
     printf "📝 %-25s: %s\n" "Alternative DNSv6 server" "2606:4700:4700::1111"
     printf -- "------------------------------------------------------------------------\n\n"
     fi
}

getroutermodel() {
    local r_ip
    r_ip=$(ip route show | grep default | awk '{print $3}' | head -n1)

    if [[ -n "$r_ip" ]]; then
        # 1. Special UPnP default paths (Port 1900/49000 combination for AVM, Telekom & others)
        local urls=(
            "http://${r_ip}:49000/igddesc.xml"
            "http://${r_ip}:49000/tr64desc.xml"
            "http://${r_ip}/rootDesc.xml"
        )
        
        local url
        for url in "${urls[@]}"; do
            local xml
            xml=$(curl -s --connect-timeout 1 "$url" 2>/dev/null)
            if [[ -n "$xml" ]]; then
                local r_model
                r_model=$(echo "$xml" | grep -oP '(?<=<modelName>)[^<]+' | head -n1)
                if [[ -n "$r_model" ]]; then
                    echo "$r_model"
                    return 0
                fi
            fi
        done
    fi
    return 1
}

setiptables() {
if command -v iptables >/dev/null 2>&1; then
     clear # Clear screen
     printf "\n\n🚀 *** Setting up security configurations for the installed programs ... ***\n\n"

     printf "\n\n➕ *** Security settings for Pi-hole DNS server\n"

     # --- Automation for iptables-persistent (for BOTH v4/v6 cases) ---
     echo iptables-persistent iptables-persistent/autosave_v4 boolean true | sudo debconf-set-selections
     echo iptables-persistent iptables-persistent/autosave_v6 boolean true | sudo debconf-set-selections

     # --- Create custom chains for IPv4 and IPv6 (if they don't exist) ---
     sudo iptables -N PIHOLE_RULES 2>/dev/null
     sudo ip6tables -N PIHOLE_RULES 2>/dev/null

     # --- Flush only the custom chains (third-party rules remain active!) ---
     sudo iptables -F PIHOLE_RULES
     sudo ip6tables -F PIHOLE_RULES

     # --- Link to the main firewall (INPUT) if necessary ---
     if ! sudo iptables -C INPUT -j PIHOLE_RULES 2>/dev/null; then
          sudo iptables -I INPUT 1 -j PIHOLE_RULES
     fi
     if ! sudo ip6tables -C INPUT -j PIHOLE_RULES 2>/dev/null; then
          sudo ip6tables -I INPUT 1 -j PIHOLE_RULES
     fi

     if ip -6 addr show 2>/dev/null | grep -q "scope global"; then
     # ******** IPv4 & IPv6 ***********
     sudo iptables -A PIHOLE_RULES -i lo -j ACCEPT # Loopback IPv4
     sudo iptables -A PIHOLE_RULES -m state --state ESTABLISHED,RELATED -j ACCEPT # Existing connections IPv4
     sudo ip6tables -A PIHOLE_RULES -i lo -j ACCEPT # Loopback IPv6
     sudo ip6tables -A PIHOLE_RULES -m state --state ESTABLISHED,RELATED -j ACCEPT # Existing connections IPv6

     # 1. FIRST: Allow access to the web interface for yourself (IMPORTANT!)
     sudo iptables -A PIHOLE_RULES -p tcp --dport 80 -j ACCEPT # for Pi-hole
     sudo ip6tables -A PIHOLE_RULES -p tcp --dport 80 -j ACCEPT # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p tcp --dport 8080 -j ACCEPT # for Pi-hole
     sudo ip6tables -A PIHOLE_RULES -p tcp --dport 8080 -j ACCEPT # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p tcp --dport 443 -j ACCEPT # for web interface
     sudo ip6tables -A PIHOLE_RULES -p tcp --dport 443 -j ACCEPT # for web interface
     
     if dpkg -l | grep -q "^ii  cups "; then
          printf "\n➕ *** Security settings for CUPS print server\n"
          sudo iptables -A PIHOLE_RULES -p tcp --dport 631 -j ACCEPT  # for print server
          sudo iptables -A PIHOLE_RULES -p udp --dport 631 -j ACCEPT  # for print server 
          sudo ip6tables -A PIHOLE_RULES -p tcp --dport 631 -j ACCEPT # for print server 
          sudo ip6tables -A PIHOLE_RULES -p udp --dport 631 -j ACCEPT # for print server 
          sudo iptables -A PIHOLE_RULES -p udp --dport 5353 -j ACCEPT # for AirPrint/mDNS
          sudo ip6tables -A PIHOLE_RULES -p udp --dport 5353 -j ACCEPT # for AirPrint/mDNS
     fi
     
        if dpkg -l | grep -q "^ii  webmin " || [ -d "/etc/webmin" ]; then
          printf "\n➕ *** Security settings for Webmin manager\n"
          sudo iptables -A PIHOLE_RULES -p tcp --dport 10000 -j ACCEPT # for Webmin manager
          sudo ip6tables -A PIHOLE_RULES -p tcp --dport 10000 -j ACCEPT # for Webmin manager
        fi

     # 2. AFTERWARDS: Block the remaining "garbage"
     sudo iptables -A PIHOLE_RULES -p tcp --destination-port 80 -j REJECT --reject-with tcp-reset # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p tcp --destination-port 8080 -j REJECT --reject-with tcp-reset # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p udp --destination-port 80 -j REJECT --reject-with icmp-port-unreachable # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p udp --destination-port 8080 -j REJECT --reject-with icmp-port-unreachable # for Pi-hole
     sudo ip6tables -A PIHOLE_RULES -p tcp --destination-port 80 -j REJECT --reject-with tcp-reset # for Pi-hole
     sudo ip6tables -A PIHOLE_RULES -p tcp --destination-port 8080 -j REJECT --reject-with tcp-reset # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p tcp --destination-port 443 -j REJECT --reject-with tcp-reset # for web interface
     sudo ip6tables -A PIHOLE_RULES -p tcp --destination-port 443 -j REJECT --reject-with tcp-reset # for web interface

     sudo iptables -A PIHOLE_RULES -i lo -j ACCEPT # Loopback IPv4
     sudo iptables -A PIHOLE_RULES -m state --state ESTABLISHED,RELATED -j ACCEPT # Existing connections IPv4
    
     sudo iptables -A PIHOLE_RULES -p tcp --dport 80 -j ACCEPT # for Pi-hole

      sudo iptables -A PIHOLE_RULES -p udp --dport 5353 -j ACCEPT # for AirPrint/mDNS
     fi
     
     if dpkg -l | grep -q "^ii  webmin " || [ -d "/etc/webmin" ]; then
       printf "\n➕ *** Security settings for Webmin manager\n"
       sudo iptables -A PIHOLE_RULES -p tcp --dport 10000 -j ACCEPT # for Webmin manager
     fi

     # 2. AFTERWARDS: Block the remaining "garbage"
     sudo iptables -A PIHOLE_RULES -p tcp --destination-port 80 -j REJECT --reject-with tcp-reset # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p tcp --destination-port 8080 -j REJECT --reject-with tcp-reset # for Pi-hole
     sudo iptables -A PIHOLE_RULES -p tcp --destination-port 443 -j REJECT --reject-with tcp-reset # for web interface
     
     if dpkg -l | grep -q "^ii  cups "; then
          sudo iptables -A PIHOLE_RULES -p tcp --destination-port 631 -j REJECT --reject-with tcp-reset # for print server

          sudo iptables -A PIHOLE_RULES -p tcp --destination-port 10000 -j REJECT --reject-with tcp-reset # for Webmin manager

     fi

     printf "\n\n⌨️ Press any key to continue...\n\n"

else

            if [[ "$choice" =~ ^[1-7]$ ]]; then
                break 
            else
                printf "\n❌ Invalid input! Please select a number from 1 to 7.\n"
                sleep 3
            fi
        done

        if (( choice > 3 )); then
            printf "\n\n⚙️ The selected upstream servers are being configured...\n"
        else
            printf "\n\n⚙️ The selected upstream server is being configured...\n"
        fi    
        
        # Declare local variables for all variants
        local cf_v4="\"1.1.1.1\", \"1.0.0.1\"" # Cloudflare IPv4
        local cf_v6="\"2606:4700:4700::1111\", \"2606:4700:4700::1001\"" # Cloudflare IPv6
        local od_v4="\"208.67.222.222\", \"208.67.220.220\"" # OpenDNS IPv4
        local od_v6="\"2620:119:35::35\", \"2620:119:53::53\"" # OpenDNS IPv6
        local q9_v4="\"9.9.9.9\", \"149.112.112.112\"" # Quad9 IPv4
        local q9_v6="\"2620:fe::fe\", \"2620:fe::9\"" # Quad9 IPv6

        # ----- IPv4 and IPv6 Upstream -----
        if ip -6 addr show 2>/dev/null | grep -q "scope global"; then
            case $choice in
                1) dns_list="[ $cf_v4, $cf_v6 ]" ;;
                2) dns_list="[ $od_v4, $od_v6 ]" ;;
                3) dns_list="[ $q9_v4, $q9_v6 ]" ;;
                4) dns_list="[ $cf_v4, $od_v4, $cf_v6, $od_v6 ]" ;;
                5) dns_list="[ $cf_v4, $q9_v4, $cf_v6, $q9_v6 ]" ;;
                6) dns_list="[ $od_v4, $q9_v4, $od_v6, $q9_v6 ]" ;;
                7) dns_list="[ $cf_v4, $od_v4, $q9_v4, $cf_v6, $od_v6, $q9_v6 ]" ;;
            esac
        else
            # ----- IPv4 only Upstream -----
            case $choice in
                1) dns_list="[ $cf_v4 ]" ;;
                2) dns_list="[ $od_v4 ]" ;;
                3) dns_list="[ $q9_v4 ]" ;;
                4) dns_list="[ $cf_v4, $od_v4 ]" ;;
                5) dns_list="[ $cf_v4, $q9_v4 ]" ;;
                6) dns_list="[ $od_v4, $q9_v4 ]" ;;
                7) dns_list="[ $cf_v4, $od_v4, $q9_v4 ]" ;;
            esac
        fi

        # Optimize Pi-hole
        printf "\n\n⚙️ Optimizing Pi-hole...\n"
        sudo pihole-FTL --config dns.cacheSize 10000 > /dev/null 2>&1
        sudo pihole-FTL --config dns.dnssec true > /dev/null 2>&1
        sudo pihole-FTL --config dns.rateLimit.count 5000 > /dev/null 2>&1
        sudo pihole-FTL --config dns.cacheLocalHosts true > /dev/null 2>&1
        sudo pihole-FTL --config dns.domainNeeded true > /dev/null 2>&1
        sudo pihole-FTL --config database.DBinterval 5 > /dev/null 2>&1
        sudo pihole-FTL --config dns.upstreams "$dns_list" > /dev/null 2>&1

        if [ -f "/etc/pihole/setupVars.conf" ]; then
         sudo sed -i '/PIHOLE_DNS_/d' /etc/pihole/setupVars.conf > /dev/null 2>&1
        fi
        printf "\n🔄 Stopping Pi-hole FTL - Please wait...\n\n"
        sudo systemctl stop pihole-FTL > /dev/null 2>&1
        sleep 3
        printf "\n⚙️ Adjusting FTL database...\n\n"
        local ddb="/etc/pihole/pihole-FTL.db"
        if [ -f "$ddb" ]; then
            # We delete the key completely instead of just updating it
            sudo sqlite3 "$ddb" "DELETE FROM config WHERE key = 'dns.upstreams';" > /dev/null 2>&1
        fi
        # Forcing the FTL database configuration
        local udb="/etc/pihole/pihole-FTL.db"
        if [ -f "$udb" ]; then
            # Briefly check if the 'config' table exists
            local table_check=$(sudo sqlite3 "$udb" "SELECT name FROM sqlite_master WHERE type='table' AND name='config';")
            if [[ -n "$table_check" ]]; then
              sudo sqlite3 "$udb" "INSERT OR REPLACE INTO config (key, value) VALUES ('dns.upstreams', '$dns_list');" > /dev/null 2>&1
            fi
        fi
        # Restart FTL
        printf "\n🔄 Restarting Pi-hole FTL - Please wait...\n\n"      
        sudo systemctl start pihole-FTL > /dev/null 2>&1
        sleep 3
        printf "\n✅ DNS Upstream servers configured and optimizations successfully completed.\n\n"
    else
        printf "\n⚠️ Pi-hole is not installed. Upstream configuration skipped.\n\n"
    fi
}

addwhitedomains() {
    if command -v pihole >/dev/null 2>&1; then
        printf "\n🚀 Adding domains that will always be allowed...\n\n"

            printf "==================================================\n"
            printf "ℹ️ The following domains will always be allowed\n"
            printf "==================================================\n"
            printf "1. Microsoft Windows Update\n"
            printf "2. YouTube\n\n"
            
        local allow_regex=(
            "(\.|^)download\.windowsupdate\.com$"
            "(\.|^)windowsupdate\.microsoft\.com$"
            "(\.|^)update\.microsoft\.com$"
            "(\.|^)download\.microsoft\.com$"
            "(\.|^)assets\.windowsupdate\.com$"
            "(\.|^)docs\.microsoft\.com$"
            "^(.+.)?(youtube\.com|googlevideo\.com|ytimg\.com|\
youtubei\.googleapis\.com|s\.youtube\.com|youtube-nocookie\.com|\
ggpht\.com|googleusercontent\.com)$"
        )

        # Direct SQL import
        local db="/etc/pihole/gravity.db"
        local sql_add="INSERT OR IGNORE INTO domainlist (type, domain, enabled, comment)"
        local added_regex_count=0
      for regex in "${allow_regex[@]}"; do
        local result
        result=$(sudo sqlite3 "$db" "$sql_add VALUES (2, '$regex', 1, 'Auto-Whitelist'); SELECT changes();")
        if [[ "$result" == "1" ]]; then
        ((added_regex_count++))
        fi
      done
       
       if [[ $added_regex_count -gt 0 ]]; then
        printf "\n\n🔄 Restarting Pi-hole FTL - Please wait...\n\n\n"
        sudo systemctl restart pihole-FTL
        sleep 5
        # Check if domains are in the database 
        # pihole-FTL-inspect domains | grep "regex"
        printf "\n✅ The allowed domains have been successfully added.\n\n"
      else
       printf "\nℹ️ All allowed domains are already present in the database.\n"
      fi
    else
        printf "\n⚠️ Pi-hole is not installed. Operation aborted!\n\n"
    fi
    printf "\n\n⌨️ Press any key to continue...\n\n"
    read -n 1 -s -r
    clear # Clear screen
}

setstaticip() {
    # Check if "sudo" is installed
    checksudo || return 1

    # If Network Manager is not installed, install it
    if ! command -v nmcli >/dev/null 2>&1; then
    printf "\n\nℹ️ The 'Network Manager' still needs to be installed - Please wait...\n\n"
    sudo dpkg --configure -a && sudo apt update && sudo apt --assume-yes upgrade && sudo apt --assume-yes dist-upgrade
    sudo apt --assume-yes autoremove
    sudo apt autoclean
    # (Note: For DEBIAN Linux only !!!)
    sudo DEBIAN_FRONTEND=noninteractive apt-get install -y -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" network-manager # Install Network Manager
    printf "\n\n🚀 The system must restart to activate the 'Network Manager'.\n\n"
    printf "\n🔄 Restarting in 10 seconds (Cancel with Ctrl+C)...\n\n"
    sleep 10
    sudo reboot
    return 1
    fi

    local skip_file="/home/$USER/.skip_ip_check"
    local sparam="$1"

    # --- Parameter check (Silent) ---
    if [[ "$1" == "-clear" ]]; then
    rm -f "$skip_file" 2>/dev/null
    fi

    # 1. Determine active interface (Language neutral)
    local interface
    interface=$(nmcli -t -f DEVICE,STATE device status | grep -E ":connected|:verbunden" | head -n 1 | cut -d: -f1)

    if [[ -z "$interface" ]]; then
        printf "\n❌ Error: No connected network interface found! - Aborting!\n\n"
        return 1
    fi

    # 2. Find the name of the active connection
    local conn_name
    conn_name=$(nmcli -t -f NAME,DEVICE connection show --active | grep ":$interface" | cut -d: -f1)

    if [[ -z "$conn_name" ]]; then
        printf "\n❌ Error: No active profile found for %s. - Aborting!\n\n" "$interface"
        return 1
    fi

    # 3. Query method via the CONNECTION
    local method
    method=$(nmcli -g ipv4.method connection show "$conn_name")
    method="${method,,}"

    # 4. Get the IP via the CONNECTION
    local current_ip=$(nmcli -g ip4.address connection show "$conn_name" | cut -d/ -f1)

    # --- Warning for Wi-Fi operation ---
    if [[ "$interface" == w* ]]; then
        printf "\n⚠️ WARNING: The DNS server is running via Wi-Fi (%s)!\n\n" "$interface"
        printf "For maximum stability and speed, a LAN cable is recommended.\n\n"
        read -r -p "❓ Continue anyway? (yes/no): " wlan_confirm
        [[ "$wlan_confirm" != "yes" ]] && { printf "\n🔄 Aborted by user.\n\n"; return 1; }
    fi

    if [[ "$method" == "manual" ]]; then
        printf "\n✅ The network interface [%s] uses a STATIC IPv4 address.\n\n" "$interface"
        return 0
    fi

    # If the flag file exists, skip the query
    if [[ -f "$skip_file" ]]; then
        return 0  # User chose DHCP
    fi

    # 3. Offer conversion
    printf "\n🌐 Current status for network interface [%s]: DHCP (Dynamic IPv4 address)\n\n" "$interface"
    printf "\n\n"
    read -r -p "❓ Would you like to switch to a STATIC IPv4 address now? (yes/no/never): " sip_antwort

    if [[ "$sip_antwort" == "yes" ]]; then
        clear # Clear screen
        printf "\n\n\n--- Configuration of the static IPv4 address ---\n\n"
        printf "\nℹ️ Note: Choose an IPv4 address outside of the DHCP pool\n"
        printf "\n👉 (e.g., FritzBox mostly: .2 to .19 or .201 to .253)\n\n\n"

        local base_ip=$(echo "$current_ip" | cut -d. -f1-3)

        # Query gateway via CONNECTION, not via DEVICE
        local current_gw=$(nmcli -g ipv4.gateway connection show "$conn_name")

        # --- Internal validation function ---
        validate_ip() {
            local ip=$1
            [[ ! "$ip" =~ ^([0-9]{1,3}\.){3}[0-9]{1,3}$ ]] && return 1
            local IFS='.'
            local oct=($ip)
            for i in {0..3}; do
                local val=$((10#${oct[$i]}))
                [[ $val -lt 0 || $val -gt 255 ]] && return 1
            done
            local last=$((10#${oct[3]}))
            [[ $last -eq 0 || $last -eq 255 ]] && return 1
            return 0
        }

        while true; do
            printf "\n\n"
            read -r -p "👉 static IPv4 address [Suggestion: ${base_ip}.212]: " new_ip
            new_ip=${new_ip:-"${base_ip}.212"}

            if ! validate_ip "$new_ip"; then
                printf "\n❌ Error: '%s' is not a valid IPv4 address (0-254, end not 0/255)!\n\n" "$new_ip"
                continue
            fi

            printf "\n🔍 Checking if IPv4 address %s is free...\n\n" "$new_ip"
            if ping -c 1 -W 1 "$new_ip" >/dev/null 2>&1; then
                printf "\n❌ OCCUPIED: Another device is already using this IPv4 address!\n\n"
            else
                break
            fi
        done

        # --- GATEWAY IPv4 address ---
        local new_gw
        local def_gw="${current_gw:-${base_ip}.1}"
        while true; do
            printf "\n\n"
            read -r -p "👉 Gateway IPv4 address (Router) - [Suggestion: $def_gw]: " new_gw
            new_gw=${new_gw:-"$def_gw"}
            # CORRECTION: ! for "If NOT valid", then error message
            if ! validate_ip "$new_gw"; then
                printf "\n❌ Error: '%s' is not a valid gateway IPv4 address!\n\n" "$new_gw"
            else
                break
            fi
        done

        # --- IMPLEMENTATION ---
        clear # Clear screen
        printf "\n\n⚙️ Settings for [%s] are being applied...\n\n" "$conn_name"

        sudo nmcli connection modify "$conn_name" \
            ipv4.addresses "${new_ip}/24" \
            ipv4.gateway "$new_gw" \
            ipv4.dns "${new_ip} 1.1.1.1" \
            ipv4.method manual \
            ipv6.method "auto" \
            ipv6.addr-gen-mode "eui64" \
            ipv6.ip6-privacy 0

        rm -f "$skip_file" 2>/dev/null
        printf "\n✅ The static IPv4 address was successfully configured: %s\n" "$new_ip"
        # ----------------------- IPv6 ----------------------------------------------
        if ip -6 addr show 2>/dev/null | grep -q "scope global"; then
         # Check if settings are applied in the Network Manager profile
         local check_v6_method=$(nmcli -g ipv6.method connection show "$conn_name")
         local check_v6_privacy=$(nmcli -g ipv6.ip6-privacy connection show "$conn_name")
         if [[ "$check_v6_method" == "auto" && "$check_v6_privacy" == "0" ]]; then
         printf "\n✅ IPv6 configuration (EUI-64) has been successfully saved.\n\n"
         else
         printf "\n⚠️ Warning: The IPv6 settings could not be verified.\n"
         fi
        fi
        # ----------------------- IPv6 ----------------------------------------------
        printf "\n⌨️ Press any key to continue...\n\n"
        read -n 1 -s -r
        printf "\n\n🚀 Restarting in 10 seconds to switch to the new IP addresses - Please wait...\n\n"
        sleep 10
        sudo reboot
        exit 0

    elif [[ "$sip_antwort" == "never" ]]; then
        printf "DNS Server - User chose a DHCP address on $(date +'%d.%m.%Y at %H:%M')!" > "$skip_file"
        printf "\n✅ The IPv4 address of the DNS server remains on DHCP configuration. (Current: %s)\n\n" "$current_ip"
        sudo nmcli connection modify "$conn_name" \
        ipv6.method "auto" \
        ipv6.addr-gen-mode "eui64" \
        ipv6.ip6-privacy 0
    else
        printf "\n🔄 Aborted. The IPv4 address remains on DHCP for now. (Current: %s)\n\n" "$current_ip"
        sudo nmcli connection modify "$conn_name" \
        ipv6.method "auto" \
        ipv6.addr-gen-mode "eui64" \
        ipv6.ip6-privacy 0
    fi
}

autoupdate() {
    # Check if "sudo" is installed
    checksudo || return 1

    local sys_path="/home/$USER/systemupdate.sh"
    local bl_path="/home/$USER/blocklistupdate.sh"
    local syslog_path="/home/$USER/systemupdate.log"
    local bllog_path="/home/$USER/blocklistupdate.log"
    local param="$1"

    # --- OPTION: DISABLE (-d) ---
    if [[ "$param" == "-d" ]]; then
        if [[ ! -f "$sys_path" && ! -f "$bl_path" ]]; then
            printf "\n⚠️ Abort: No 'Autoupdate' files found.\n\n"
            return 1
        fi

        read -r -p "❓ Do you really want to disable auto-updates? (yes/no): " auconfirm
        if [[ "$auconfirm" == "yes" ]]; then
            rm -f "$sys_path" "$syslog_path" "$bl_path" "$bllog_path"
            (crontab -l 2>/dev/null | grep -vE "($sys_path|$bl_path)") | crontab -
            printf "\n✅ Auto-update has been disabled.\n\n"
        else
            printf "\n🔄 The process was aborted.\n\n"
        fi
        return 0
    fi

    # --- OPTION: CREATE (-c) ---
    if [[ "$param" == "-c" ]]; then
        local run_sys="true"
        local run_bl="true"

        # Check System Update
        if [[ -f "$sys_path" ]]; then
            printf "\n"
            read -r -p "⚠️ The 'Autoupdate' (System) already exists. Overwrite? (yes/no): " overwrite_sys
            if [[ "$overwrite_sys" != "yes" ]]; then
                printf "\n⚠️ Abort: System update script will not be overwritten.\n\n"
                run_sys="false"
            fi
        fi
        
        # Check Blocklists
        if [[ -f "$bl_path" ]]; then
            printf "\n"
            read -r -p "⚠️ The 'Autoupdate' (Blocklists) already exists. Overwrite? (yes/no): " overwrite_bl
            if [[ "$overwrite_bl" != "yes" ]]; then
                printf "\n⚠️ Abort: Blocklist update script will not be overwritten.\n\n"
                run_bl="false"
            fi
        fi

        # If both are "false", we can stop here completely
        if [[ "$run_sys" == "false" && "$run_bl" == "false" ]]; then
            return 1
        fi

        # 1. Create System Update script (only if run_sys=true)
        if [[ "$run_sys" == "true" ]]; then
            cat <<-EOF > "$sys_path"
#!/bin/bash
rm -f "$syslog_path"
date > "$syslog_path" 2>&1
sudo pihole disable >> "$syslog_path" 2>&1
sudo apt-get update >> "$syslog_path" 2>&1
sudo apt-get --assume-yes upgrade >> "$syslog_path" 2>&1
sudo apt-get --assume-yes dist-upgrade >> "$syslog_path" 2>&1
sudo apt-get --assume-yes upgrade --fix-missing >> "$syslog_path" 2>&1
sudo apt-get --assume-yes autoremove >> "$syslog_path" 2>&1
sudo apt-get autoclean >> "$syslog_path" 2>&1
sudo pihole -up >> "$syslog_path" 2>&1
sudo pihole enable >> "$syslog_path" 2>&1
echo "***Reboot after system update***" >> "$syslog_path" 2>&1
sudo reboot >> "$syslog_path" 2>&1
EOF
            chmod +x "$sys_path"
        fi
# =======================================================================
        # 2. Create Blocklist Update script (only if run_bl=true)
        if [[ "$run_bl" == "true" ]]; then
            cat <<-EOF > "$bl_path"
#!/bin/bash
rm -f "$bllog_path"
date > "$bllog_path" 2>&1
sudo pihole disable >> "$bllog_path" 2>&1
sudo apt-get update >> "$bllog_path" 2>&1
sudo apt-get --assume-yes upgrade >> "$bllog_path" 2>&1
sudo apt-get --assume-yes dist-upgrade >> "$bllog_path" 2>&1
sudo apt-get --assume-yes upgrade --fix-missing >> "$bllog_path" 2>&1
sudo apt-get --assume-yes autoremove >> "$bllog_path" 2>&1
sudo apt-get autoclean >> "$bllog_path" 2>&1
sudo pihole -g >> "$bllog_path" 2>&1
sudo pihole enable >> "$bllog_path" 2>&1
echo "***Reboot after blocklist update***" >> "$bllog_path" 2>&1
sudo reboot >> "$bllog_path" 2>&1
EOF
            chmod +x "$bl_path"
        fi

        # --- UPDATE CRONTAB ---
        {

local_lang_en() {
    # Check if "sudo" is installed
    checksudo || return 1

    # --- CHECK: Is English already active? ---
    local kbd_check=0
    local lang_check=0

    # Check for US keyboard layout
    if grep -q 'XKBLAYOUT="us"' /etc/default/keyboard 2>/dev/null; then
        kbd_check=1
    fi

    # Check for US English locale
    if [[ "$LANG" == "en_US.UTF-8" ]]; then
        lang_check=1
    fi

    if [[ $kbd_check -eq 0 || $lang_check -eq 0 ]]; then
        # Ensure a clean environment for the process
        export LC_ALL=C.UTF-8
        export LANG=C.UTF-8
        export LANGUAGE=C.UTF-8
        printf "\n🌐 Switching system language and keyboard to English (US) - Please wait...\n\n\n"

        # Write keyboard configuration
        cat <<-EOF | sudo tee /etc/default/keyboard > /dev/null
XKBMODEL="pc105"
XKBLAYOUT="us"
XKBVARIANT=""
XKBOPTIONS=""
BACKSPACE="guess"
EOF
        # Disable all other languages (comment them out)
        sudo sed -i 's/^[^#]/# &/g' /etc/locale.gen

        # Add English (US)
        echo "en_US.UTF-8 UTF-8" | sudo tee -a /etc/locale.gen > /dev/null
        
        # Write system locale files (takes effect after reboot)
        echo "LANG=en_US.UTF-8" | sudo tee /etc/default/locale > /dev/null
        echo "LC_ALL=en_US.UTF-8" | sudo tee -a /etc/default/locale > /dev/null
        
        # Generate the locale physically
        sudo locale-gen en_US.UTF-8 > /dev/null 2>&1
        
        # Officially register in the system
        sudo update-locale LANG=en_US.UTF-8 LC_ALL=en_US.UTF-8
        
        # Activate keyboard for the current session
        sudo loadkeys us 2>/dev/null
        
        # Set timezone (Optional: changed to UTC or keep as needed. Here set to UTC for English default)
        if [[ "$(cat /etc/timezone 2>/dev/null)" != "UTC" ]]; then
            sudo timedatectl set-timezone UTC 2>/dev/null
        fi
        
        printf "\n✅ System language and keyboard successfully switched to English.\n\n"
        return 1
    else
        printf "\nℹ️ System language and keyboard are already set to English.\n\n"
        return 0
    fi
}

setfail2banjail() {
    printf "\n🛡️ Configuring Fail2Ban protection (dynamic detection)...\n\n"
    
    local my_ip=$(hostname -I | awk '{print $1}')

    # --- Check: What is actually installed and active? ---
    local apache_active="false"
    local lighttpd_active="false"
    local webmin_active="false"
    local ssh_active="false"

    # Check if services/directories exist
    command -v apache2 >/dev/null 2>&1 && apache_active="true"
    command -v lighttpd >/dev/null 2>&1 && lighttpd_active="true"
    [[ -d "/etc/webmin" ]] && webmin_active="true"
    
    # Check if SSH is active in the system
    systemctl is-active --quiet ssh && ssh_active="true"

    # 1. File: fail2ban.local (IPv6 Support - only if IPv6 is enabled)
    if ip -6 addr show 2>/dev/null | grep -q "scope global"; then
      local f2b_conf="/etc/fail2ban/fail2ban.local"
      if ! grep -q "allowipv6 = auto" "$f2b_conf" 2>/dev/null; then
        printf "➕ Configuring IPv6 support...\n"
        echo -e "[DEFAULT]\nallowipv6 = auto" | sudo tee "$f2b_conf" > /dev/null
      fi
    fi

    # 2. File: jail.local (Central configuration)
    printf "➕ Creating jail file: 'jail.local'...\n"
    sudo cat <<-EOF | sudo tee /etc/fail2ban/jail.local > /dev/null
[DEFAULT]
ignoreip = 127.0.0.1/8 ::1 $my_ip
bantime  = 86400
maxretry = 5
backend  = systemd

[sshd]
enabled = $ssh_active
port    = ssh
filter  = sshd

[apache-auth]
enabled = $apache_active
filter  = apache-auth
port    = http,https
logpath = /var/log/apache2/error.log

[lighttpd-custom]
enabled = $lighttpd_active
port    = http,https
logpath = /var/log/lighttpd/error.log
filter  = lighttpd-custom
EOF

    # 3. Webmin Manager Jail logic (Only if Webmin is active)
    if [[ "$webmin_active" == "true" ]]; then
        printf "➕ Creating jail file 'webmin.conf'...\n"
        sudo cat <<-EOF | sudo tee /etc/fail2ban/jail.d/webmin.conf > /dev/null
[webmin-auth]
enabled = $webmin_active
port    = 10000
filter  = webmin-auth
backend = systemd
EOF
    else
        sudo rm -f /etc/fail2ban/jail.d/webmin.conf 2>/dev/null
        printf "\n🧹 Graphical interface 'Webmin Manager' not found – removing jail file!\n"
    fi

    # Restart and status output
    printf "\n🔄 Restarting 'Fail2Ban' service...\n\n"
    sudo systemctl restart fail2ban
    sleep 5
    if systemctl is-active --quiet fail2ban; then
        printf "\n✅ Fail2Ban configuration completed:\n"
        printf "👉 - SSH:       $([[ "$ssh_active" == "true" ]] && echo "ACTIVE ✅" || echo "INACTIVE ❌")\n"
        printf "👉 - Webmin:    $([[ "$webmin_active" == "true" ]] && echo "ACTIVE ✅" || echo "INACTIVE ❌")\n"
        printf "👉 - Webserver: $([[ "$apache_active" == "true" || "$lighttpd_active" == "true" ]] && echo "ACTIVE ✅" || echo "INACTIVE ❌")\n\n"
    else
        printf "\n❌ Error starting 'Fail2Ban' - Aborting!\n\n"
    fi
 printf "\n⌨️ Press any key to continue...\n\n"
 read -n 1 -s -r
 clear # Clear screen
}

storagestatus() {
    local param="$1"
    if command -v pihole >/dev/null 2>&1; then
        printf "\n\n"
        printf "===========================================\n"
        printf "🗄️  Pi-hole STORAGE STATUS\n"
        printf "==========================================="

        # 1. Determine root partition and drive
        local root_part=$(findmnt -n -o SOURCE /)
        local parent_disk=$(lsblk -no PKNAME "$root_part")
        
        # 2. Disk info and database sizes
        local fmt='s/\([0-9]\)\([GKkM]\)/\1 \2B/'
        local disk_usage=$(df -h / | awk 'NR==2 {print $5}' | sed 's/%/ %/')
        
        local g_db="/etc/pihole/gravity.db"
        local f_db="/etc/pihole/pihole-FTL.db"
        local gravity_size=$(sudo du -sh "$g_db" 2>/dev/null | awk '{print $1}' | sed "$fmt")
        local ftl_size=$(sudo du -sh "$f_db" 2>/dev/null | awk '{print $1}' | sed "$fmt")

        # 3. Formatted output with fixed width 
        printf "\n💾 %-22s: %s (%s)\n" "Drive" "/dev/$parent_disk" "${root_part#/dev/}"
        printf "📍 %-22s: %s\n" "Disk usage at" "$disk_usage"
        printf "📝 %-22s: %s\n" "Blocklist database" "${gravity_size:-0 B}"
        printf "📈 %-22s: %s\n" "Statistics database" "${ftl_size:-0 B}"
        printf -- "-------------------------------------------\n\n"
    else
        if [[ "$param" != "-phc" ]]; then
            printf "\n⚠️  Pi-hole is not installed. Storage status cannot be displayed! \n\n"
        fi
    fi
}

clearpiholelogs() {
    # Check: Is Pi-hole installed?
    if command -v pihole >/dev/null 2>&1; then
    printf "\n"
    printf "\n⚠️ WARNING: This will permanently delete the entire query log and all\n"
    printf "   statistics (graphs/history)!\n\n"

    read -r -p "❓ Do you really want to delete the Pi-hole logs and statistics? (yes/no): " logconfirm

    if [[ "$logconfirm" == "yes" ]]; then
        printf "\n🧹 Clearing Pi-hole logs and statistics - Please wait...\n\n"

        # 1. Stop service
        printf "\n🔄 Stopping Pi-hole FTL - Please wait...\n\n\n"
        sudo systemctl stop pihole-FTL > /dev/null 2>&1
        sleep 5

        # 2. Delete the statistics database
        if [ -f "/etc/pihole/pihole-FTL.db" ]; then
            sudo rm /etc/pihole/pihole-FTL.db
            printf "\n✅ The statistics database has been deleted.\n"
        fi

        # 3. Clear the text log
        if [ -f "/var/log/pihole/pihole.log" ]; then
            sudo truncate -s 0 /var/log/pihole/pihole.log
            printf "\n✅ The Pi-hole text log has been cleared.\n"
        fi

        # 4. Start service again
        printf "\n🔄 Starting Pi-hole FTL - Please wait...\n\n\n"
        sudo systemctl start pihole-FTL > /dev/null 2>&1
        sleep 5
        printf "✨ All logs and statistics have been successfully deleted - Please wait...\n\n"
        if systemctl is-active --quiet pihole-FTL; then
        storagestatus # Display storage status
        fi
    else
        printf "\n🔄 Operation aborted. Logs and statistics have been preserved.\n\n"
    fi
    else 
    printf "\n⚠️ Pi-hole is not installed. Logs and statistics cannot be deleted! \n\n"
    fi
}
